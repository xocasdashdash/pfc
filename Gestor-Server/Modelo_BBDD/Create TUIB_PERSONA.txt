CREATE TABLE UXXIAC.TUIB_PERSONA
(

  DNIPRS         VARCHAR2(15 BYTE) CONSTRAINT RUIB_NPRS_DNIPRS NOT NULL,

  MUN_CODNUMNAI  NUMBER(10),

  DNIPAE         VARCHAR2(15 BYTE),
  TIPDCMIDEPAE   VARCHAR2(1 BYTE),

  LL2PAE         VARCHAR2(32 BYTE),
  DNIMAE         VARCHAR2(15 BYTE),
  TIPDCMIDEMAE   VARCHAR2(1 BYTE),
  NOMMAE         VARCHAR2(32 BYTE),

COMMENT ON COLUMN UXXIAC.TUIB_PERSONA.TLFMOV IS 'Teléfono móvil de la persona';

COMMENT ON COLUMN UXXIAC.TUIB_PERSONA.IDG_RIU IS 'Identificador de persona en RIU';

COMMENT ON COLUMN UXXIAC.TUIB_PERSONA.PORCDIS IS 'Porcentaje que indica el grado de discapacidad';

COMMENT ON COLUMN UXXIAC.TUIB_PERSONA.NUMCARNETDIS IS 'Número del carnet de discapacidad';
